class CStack
{
    char *s;
    int tp;
    int size;
public:
    CStack(int initSize = 5)
    {
        size = initSize;
        s = new char[size];
        tp = -1;

    }
    ~CStack()
    {
        delete[] s;
    }
    bool isEmpty()
    {
        return tp == -1;

    }
    bool isFull()
    {
        return tp == size - 1;

    }
    void push(char c)
//入栈
    {
        if(isFull())
        {
            char *temp = new char[size * 2];
            for(int i = 0; i < size; i  ++)
            {
                temp[i] = s[i];
            }
            delete[] s;
            s = temp;
            size *= 2;
        }
        s[++tp] = c;

    }
    char pop()
//出栈，
    {
        if(tp >= 0)
        {
            return s[tp --];
        }
        else
            return '\0';
    }
    char top()
//返回栈顶元素，初始为-1
    {
        if(tp >= 0)
            return s[tp];
        else
            return '\0';
    }
};



int main()
{
    string str;
    getline(cin, str);
    
    CStack sss;
    for(int i = 0; i < str.size(); i ++)
    {
        if(str[i] == '{' || str[i] == '[' || str[i] == '(')
            sss.push(str[i]);
        if(str[i] == ')')
        {
            char result = sss.pop();
            if(result != '(')
            {
                cout << "Not balanced" << endl;
                return 0;
            }
        }
        if(str[i] == ']')
        {
            char result = sss.pop();
            if(result != '[')
            {
                cout << "Not balanced" << endl;
                return 0;
            }
        }
        if(str[i] == '}')
        {
            char result = sss.pop();
            if(result != '{')
            {
                cout << "Not balanced" << endl;
                return 0;
            }
        }
        
        
    }
//匹配逻辑，应当检查是否为空
    if(!sss.isEmpty())
    {
        cout << "Not balanced" << endl;
    }
    else
    {
        cout << "Balanced" << endl;
    }
    return 0;
}

        cout << endl;
        cout << "Player " << player << " points: " << score << endl;

        return score;
    }
};

#include <iostream> 
#include <cmath>
#include <iomanip>

using namespace std; 
 
class CPoint
{
protected:
    double x, y;
public:
    CPoint(){}
// 这里可以初始化为0
    CPoint(double a, double b) : x(a), y(b){}
    CPoint(const CPoint& other)
//浅拷贝可以通过编译器自动生成，深拷贝才一定要写
    {
        x = other.getx();
        y = other.gety();
    }
    
    double getDist(const CPoint& other)
    {
        double distx = (x - other.getx());
        double disty = (y - other.gety());
        distx *= distx;
        disty *= disty;
        
        return sqrt(distx + disty);
    }
    
    double getx() const
    {
        return x;
    }
    double gety() const
    {
        return y;
    }
    void show()
    {
        cout << "(" << x << ", " << y << ")" << endl;
    }
    friend istream& operator>>(istream& in, CPoint &c);
};

istream& operator>>(istream& in, CPoint &c)
{
    double a, b;
    in >> a >> b;
    c.x = a;
    c.y = b;
    return in;
}
class CTriangle 
{
private:
    CPoint c1, c2, c3;
    double a, b, c;
public:
    CTriangle(const CPoint& o1, const CPoint& o2, const CPoint& o3) : c1(o1), c2(o2), c3(o3) {
        a = c1.getDist(c2);
        b = c1.getDist(c3);
        c = c2.getDist(c3);
    }
    
    double getArea()
    {
        double p = getC() / 2;
        return sqrt(p * (p - a) * (p - b) * (p - c));
    }
    double getC()
    {
        return a + b + c;
    }
};
 
 
int main() 
{ 
    CPoint c1, c2, c3;
    cin >> c1 >> c2 >> c3;
    CTriangle t(c1, c2, c3);
    cout << setprecision(10)<< t.getC() << ' ' << t.getArea() << endl;
}

<summary> 友元函数访问权限</summary>

| 项目              | 友元函数        |
| --------------- | ----------- |
| 是类的成员吗          | ❌ 不是        |
| 有 `this` 指针吗    | ❌ 没有        |
| 能访问 `public`    | ✅           |
| 能访问 `protected` | ✅           |
| 能访问 `private`   | ✅           |
| 能调用成员函数         | ✅（通过对象）     |
| 能直接使用成员变量名      | ❌           |
| 调用方式            | `func(obj)` |
| 定义位置            | 类内声明，类外实现   |

</details>
