

## 动态堆栈

```c++
class CStack
{
    char *s;
    int tp;
    int size;
public:
    CStack(int initSize = 5)
    {
        size = initSize;
        s = new char[size];
        tp = -1;

    }
    ~CStack()
    {
        delete[] s;
    }
    bool isEmpty()
    {
        return tp == -1;

    }
    bool isFull()
    {
        return tp == size - 1;

    }
    void push(char c)
//入栈
    {
        if(isFull())
        {
            char *temp = new char[size * 2];
            for(int i = 0; i < size; i  ++)
            {
                temp[i] = s[i];
            }
            delete[] s;
            s = temp;
            size *= 2;
        }
        s[++tp] = c;

    }
    char pop()
//出栈，
    {
        if(tp >= 0)
        {
            return s[tp --];
        }
        else
            return '\0';
    }
    char top()
//返回栈顶元素，初始为-1
    {
        if(tp >= 0)
            return s[tp];
        else
            return '\0';
    }
};



int main()
{
    string str;
    getline(cin, str);
    
    CStack sss;
    for(int i = 0; i < str.size(); i ++)
    {
        if(str[i] == '{' || str[i] == '[' || str[i] == '(')
            sss.push(str[i]);
        if(str[i] == ')')
        {
            char result = sss.pop();
            if(result != '(')
            {
                cout << "Not balanced" << endl;
                return 0;
            }
        }
        if(str[i] == ']')
        {
            char result = sss.pop();
            if(result != '[')
            {
                cout << "Not balanced" << endl;
                return 0;
            }
        }
        if(str[i] == '}')
        {
            char result = sss.pop();
            if(result != '{')
            {
                cout << "Not balanced" << endl;
                return 0;
            }
        }
        
        
    }
//匹配逻辑，应当检查是否为空
    if(!sss.isEmpty())
    {
        cout << "Not balanced" << endl;
    }
    else
    {
        cout << "Balanced" << endl;
    }
    return 0;
}
```
